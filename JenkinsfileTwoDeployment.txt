pipeline{
    agent any
    environment {
    FIREBASE_DEPLOY_TOKEN = credentials('firebase-token')
    }

    stages{
        stage('Building'){
            steps{	
            sh 'curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.1/install.sh | bash ; source ~/.bashrc; nvm install --lts; sudo npm install -g firebase-tools'
            }
        }
        stage('Staging'){
            steps{
            sh 'firebase deploy -P staging-replica-c0b2f --token "$FIREBASE_DEPLOY_TOKEN"'
            }
        }
        stage('Production'){
            steps{
            sh 'firebase deploy -P production-replica-20392 --token "$FIREBASE_DEPLOY_TOKEN"'
            }
        }
    }
}
